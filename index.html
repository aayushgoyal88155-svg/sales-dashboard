<script>

let data=[]

const sheetURL="https://script.google.com/macros/s/AKfycbw9TeE6b_BrF87bVy1GhKEjo6EK_jvXJBM44A8hX1OQHI9p56YzEynrWBuWbUh7_bjIEQ/exec"

function loadSheetData(){

fetch(sheetURL)
.then(res=>res.json())
.then(sheetData=>{

data=[]

sheetData.forEach(r=>{

data.push({
date:r[0],
agent:(r[1] || "").toString().trim(),
calls:parseInt(r[2]) || 0,
orders:parseInt(r[3]) || 0,
sales:parseInt(r[4]) || 0
})

})

updateDashboard(data)

})
.catch(err=>{
console.error("Error loading sheet:",err)
})

}

document.getElementById('salesForm').addEventListener('submit',function(e){

e.preventDefault()

let date=document.getElementById('date').value
let agent=document.getElementById('agent').value
let calls=parseInt(document.getElementById('calls').value)
let orders=parseInt(document.getElementById('orders').value)
let sales=parseInt(document.getElementById('sales').value)

let duplicate=data.find(d=>d.date===date && d.agent===agent)

if(duplicate){
alert("Entry already exists for this agent today")
return
}

fetch(sheetURL,{
method:"POST",
body:JSON.stringify({
date:date,
agent:agent,
calls:calls,
orders:orders,
sales:sales
})
})
.then(()=>loadSheetData())

})

function updateDashboard(dataset){

let tbody=document.querySelector('#dataTable tbody')
tbody.innerHTML=""

let totalCalls=0
let totalOrders=0
let totalSales=0

let agentStats={
Palak:{calls:0,orders:0},
Sandeep:{calls:0,orders:0}
}

dataset.forEach(d=>{

tbody.innerHTML+=`
<tr>
<td>${d.date}</td>
<td>${d.agent}</td>
<td>${d.calls}</td>
<td>${d.orders}</td>
<td>${d.sales}</td>
</tr>
`

totalCalls+=d.calls
totalOrders+=d.orders
totalSales+=d.sales

if(agentStats[d.agent]){
agentStats[d.agent].calls+=d.calls
agentStats[d.agent].orders+=d.orders
}

})

document.getElementById('totalCalls').innerText=totalCalls
document.getElementById('totalOrders').innerText=totalOrders
document.getElementById('totalSales').innerText="₹"+totalSales

let aov = totalOrders ? (totalSales/totalOrders).toFixed(0) : 0
document.getElementById('aov').innerText="₹"+aov

let palakConv = agentStats.Palak.calls ? ((agentStats.Palak.orders/agentStats.Palak.calls)*100).toFixed(1) : 0
let sandeepConv = agentStats.Sandeep.calls ? ((agentStats.Sandeep.orders/agentStats.Sandeep.calls)*100).toFixed(1) : 0

document.getElementById('palakConv').innerText=palakConv+"%"
document.getElementById('sandeepConv').innerText=sandeepConv+"%"

let avgCalls=((agentStats.Palak.calls+agentStats.Sandeep.calls)/2).toFixed(0)
document.getElementById('avgCalls').innerText=avgCalls

}

function filterDay(){

let selected=document.getElementById("filterDate").value
let filtered=data.filter(d=>d.date===selected)
updateDashboard(filtered)

}

loadSheetData()
